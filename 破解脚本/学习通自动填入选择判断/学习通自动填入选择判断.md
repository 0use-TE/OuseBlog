function setAnswers(str) {
  const elements = document.getElementsByClassName('stem_answer');
  console.log(`找到 ${elements.length} 个 stem_answer 元素`);
  
  str.split('|').forEach((item, i) => {
    const parts = item.split(',');
    const index = parts.length === 2 ? parseInt(parts[0]) - 1 : i;
    const value = parts.length === 2 ? parts[1] : parts[0];
    const input = elements[index]?.previousElementSibling;
    
    if (input) {
      input.value = value;
      console.log(`第 ${index + 1} 题设置为: ${value}`);
    } else {
      console.warn(`未找到第 ${index + 1} 题的 input`);
    }
  });

  console.log("全部答案已设置完成。");
}
setAnswers('');
